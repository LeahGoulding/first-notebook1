#read file into R 
#assuming have set up working directory with the data folder using following command
pop_data<- read_csv("data/gapminder_data.csv")

# create a table , e.g. with data from European countries in 2007 - ordered by life expectancy. 

euro_data_table <- pop_data %>%
  filter(continent == "Europe" & year ==2007) %>% 
  select(-continent, -year) %>% 
  arrange(desc(lifeExp)) %>% 
  rename("Population Size" = pop, "Life Expectancy"  = lifeExp, "GDP" = gdpPercap)

#removing continent and year column as they will all be Europe and 2007 so those columns in the table would be useful

Adding Images

euro_data_fig <- pop_data %>% 
  filter(continent == "Europe") %>% 
  mutate(year=as_factor(year))

euro_plot <- euro_data_fig %>% 
  ggplot(mapping = aes(x=year, y=lifeExp)) +
  geom_violin() +
  stat_summary(fun = median, geom = "point" , colour = "red")

euro_plot

# for compiling a figure using multiple images use ggarrange()
# keep only UK data 

uk_data_fig<- pop_data %>% 
  filter(country=="United Kingdom") %>% 
  mutate(year=as_factor(year))

Now we will display more than one image together

# UK plot

uk_plot <- uk_data_fig %>% 
  ggplot(mapping = aes(x=year, y=lifeExp)) +
  geom_point()

ggarrange(euro_plot, uk_plot, ncol = 2, nrow = 1, labels =  "AUTO", common.legend = TRUE)
